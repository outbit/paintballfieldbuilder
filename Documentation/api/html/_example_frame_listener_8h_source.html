00308         // Override frameRenderingQueued event to process that (don't care about frameEnded)
00309         bool frameRenderingQueued(const FrameEvent& evt)
00310         {
00311 
00312                 if(mWindow->isClosed()) return false;
00313 
00314                 mSpeedLimit = mMoveScale * evt.timeSinceLastFrame;
00315 
00316                 //Need to capture/update each device
00317                 mKeyboard->capture();
00318                 mMouse->capture();
00319                 if( mJoy ) mJoy->capture();
00320 
00321                 bool buffJ = (mJoy) ? mJoy->buffered() : true;
00322 
00323         Ogre::Vector3 lastMotion = mTranslateVector;
00324 
00325                 //Check if one of the devices is not buffered
00326                 if( !mMouse->buffered() || !mKeyboard->buffered() || !buffJ )
00327                 {
00328                         // one of the input modes is immediate, so setup what is needed for immediate movement
00329                         if (mTimeUntilNextToggle >= 0)
00330                                 mTimeUntilNextToggle -= evt.timeSinceLastFrame;
00331 
00332                         // Move about 100 units per second
00333                         mMoveScale = mMoveSpeed * evt.timeSinceLastFrame;
00334                         // Take about 10 seconds for full rotation
00335                         mRotScale = mRotateSpeed * evt.timeSinceLastFrame;
00336 
00337                         mRotX = 0;
00338                         mRotY = 0;
00339                         mTranslateVector = Ogre::Vector3::ZERO;
00340 
00341                 }
00342 
00343                 //Check to see which device is not buffered, and handle it
00344                 if( !mKeyboard->buffered() )
00345                         if( processUnbufferedKeyInput(evt) == false )
00346                                 return false;
00347                 if( !mMouse->buffered() )
00348                         if( processUnbufferedMouseInput(evt) == false )
00349                                 return false;
00350 
00351                 // ramp up / ramp down speed
00352         if (mTranslateVector == Ogre::Vector3::ZERO)
00353                 {
00354                         // decay (one third speed)
00355                         mCurrentSpeed -= evt.timeSinceLastFrame * 0.3;
00356                         mTranslateVector = lastMotion;
00357                 }
00358                 else
00359                 {
00360                         // ramp up
00361                         mCurrentSpeed += evt.timeSinceLastFrame;
00362 
00363                 }
00364                 // Limit motion speed
00365                 if (mCurrentSpeed > 1.0)
00366                         mCurrentSpeed = 1.0;
00367                 if (mCurrentSpeed < 0.0)
00368                         mCurrentSpeed = 0.0;
00369 
00370                 mTranslateVector *= mCurrentSpeed;
00371 
00372 
00373                 if( !mMouse->buffered() || !mKeyboard->buffered() || !buffJ )
00374                         moveCamera();
00375 
00376                 return true;
00377         }
